深度优先搜索（使用递归）
```cpp
class Solution {
public:
    vector<bool> vcRes;
    int num=0;

    void dfs(vector<vector<int>>& rooms, int x)
    {
        vcRes[x] = true;
        num++;
        for(auto &key : rooms[x])
        {
            if(!vcRes[key])
            {
                dfs(rooms, key);
            }
        }
    }
     
    bool canVisitAllRooms(vector<vector<int>>& rooms) {
        int n = rooms.size();
        vcRes.resize(n);
        dfs(rooms, 0);
        if(num == n)
            return true;
        else
            return false;
    }
};
```


广度优先搜索(使用队列）
```cpp
class Solution {
public:     
    bool canVisitAllRooms(vector<vector<int>>& rooms) {
        int n = rooms.size();
        vector<bool> vcRes(n);
        int num = 0;
        queue<int> dp;
        dp.emplace(0);
        while(!dp.empty())
        {
            int loc = dp.front();
            dp.pop();
            if(!vcRes[loc])
            {
                vcRes[loc] = true;
                num ++;
                for(auto x : rooms[loc])
                {
                    dp.emplace(x);
                }
            }
            
        }
        if(num == n)
            return true;
        else
            return false;
    }
};
```
